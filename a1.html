Firmware!

        // the setup function runs once when you press reset or power the board
        void setup() {
          // initialize digital pin LED_BUILTIN as an output.
          pinMode(LED_BUILTIN, OUTPUT);     // initialize pin 13 output
          pinMode(12, OUTPUT);              // initialize pin 12 output
          pinMode(8, OUTPUT);               // initialize pin 8 output
        }

        // the loop function runs over and over again forever
        void loop() {
          int i = 0;
          for (i = 0; i < 1; i++) {           // blink the red LED for 3 counts
            digitalWrite(LED_BUILTIN, HIGH);  // turn the LED on (HIGH is the voltage level)
            delay(500);                       // wait for a second
            digitalWrite(LED_BUILTIN, LOW);   // turn the LED off by making the voltage LOW
            delay(500);                       // wait for a second
          }
        
          for (i = 1; i < 2; i++) {          // blink the green LED for 3 counts
            digitalWrite(12, HIGH);          // turn the LED on (HIGH is the voltage level)
            delay(500);                      // wait for a second
            digitalWrite(12, LOW);           // turn the LED off by making the voltage LOW
            delay(500);                      // wait for a second
          }
        
          for (i = 2; i < 3; i++) {          // blink the blue LED for 3 counts
            digitalWrite(8, HIGH);           // turn the LED on (HIGH is the voltage level)
            delay(500);                      // wait for a second
            digitalWrite(8, LOW);            // turn the LED off by making the voltage LOW
            delay(500);                      // wait for a second
          }

          i = 0;                            // reset the counter to restart the blinking loop
        }
